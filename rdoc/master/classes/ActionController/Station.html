records_from_path(options = {})

Obtains all ActiveRecord record in parameters.

Given this URI: 

  /projects/1/tasks/2/posts/3

  records_from_path #=> [ Project-1, Task-2 ]

Options:

  • acts_as: the ActiveRecord model must acts_as the given symbol. acts_as => :container

[ show source ]

67:     def records_from_path(options = {})
68:       acts_as_module = "ActiveRecord::#{ options[:acts_as].to_s.classify }".constantize if options[:acts_as]
69: 
70:       candidates = params.keys.select{ |k| k[-3..-1] == '_id' }
71: 
72:       candidates.map { |candidate_key|
73:         # Filter keys that correspond to classes
74:         begin
75:           candidate_class = candidate_key[0..-4].to_sym.to_class
76:         rescue NameError
77:           next
78:         end
79: 
80:         # acts_as filter
81:         if options[:acts_as]
82:           next unless acts_as_module.classes.include?(candidate_class)
83:         end
84: 
85:         next unless candidate_class.respond_to?(:find)
86: 
87:         # Find record
88:         begin
89:           candidate_class.find_with_param(params[candidate_key])
90:         rescue ::ActiveRecord::RecordNotFound
91:           next
92:         end
93:       }.compact
94:     end
