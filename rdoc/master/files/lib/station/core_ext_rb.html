62:       xml = options[:xml] || eval("xml", block.binding)
63:       xml.instruct!
64: 
65:       xml.service "xmlns" => "http://www.w3.org/2007/app", "xmlns:atom" => 'http://www.w3.org/2005/Atom' do
66:       # Workspaces are Containers current_agent can post to:
67:       record.stages.select{ |s| s.authorize?(:read, :to => current_agent) }.each do |container|
68:         xml.workspace do
69:           xml.tag!( "atom:title", container.name )
70:             if container.authorize?([ :read, :Content ], :to => current_agent)
71:               # Collections are different type of Contents
72:               for content in container.accepted_content_types
73:                 xml.collection(:href => polymorphic_path([ container, content.to_class.new ]) + '.atom') do
74:                   xml.tag!("atom:title", I18n.t('other_in_container', :container => container.name, :scope => content.to_class.to_s.underscore))
75:                   xml.accept(container.authorize?([ :create, :Content ], :to => current_agent) ? content.to_class.accepts : nil)
76:                 end
77:               end
78:             end
79:           end
80:         end
81:       end
